USE railway; -- Isso seleciona o banco correto(ou seta ele como padrão)

-- 1. CRIAÇÃO DA TABELA DE PRODUTOS
CREATE TABLE produtos (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(100) NOT NULL,
    descricao TEXT,
    preco DECIMAL(10,2) NOT NULL,
    categoria ENUM('Pizza Salgada', 'Pizza Doce', 'Porção', 'Bebida', 'Sobremesa') NOT NULL,
    disponivel BOOLEAN DEFAULT TRUE
);

-- 2. CRIAÇÃO DA TABELA DE RESERVAS
CREATE TABLE reservas (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nome_cliente VARCHAR(100) NOT NULL,
    telefone_cliente VARCHAR(20),
    data_reserva DATE NOT NULL,
    hora_reserva TIME NOT NULL,
    quantidade_pessoas INT NOT NULL,
    status_reserva ENUM('Pendente', 'Confirmada', 'Cancelada', 'Finalizada') DEFAULT 'Pendente'
);

-- 3. CRIAÇÃO DA TABELA DE PEDIDOS
CREATE TABLE pedidos (
    id INT PRIMARY KEY AUTO_INCREMENT,
    data_pedido DATETIME DEFAULT CURRENT_TIMESTAMP,
    cliente_nome VARCHAR(100) NOT NULL,
    tipo_pedido ENUM('Delivery', 'Retirada', 'Mesa') NOT NULL,
    endereco_entrega VARCHAR(255),
    valor_total DECIMAL(10,2) DEFAULT 0.00,
    status_pedido ENUM('Recebido', 'Na Cozinha', 'Em Entrega', 'Concluído') DEFAULT 'Recebido'
);

-- 4. CRIAÇÃO DA TABELA DE ITENS DO PEDIDO
CREATE TABLE itens_pedido (
    id INT PRIMARY KEY AUTO_INCREMENT,
    pedido_id INT,
    produto_id INT,
    quantidade INT NOT NULL,
    subtotal DECIMAL(10,2),
    FOREIGN KEY (pedido_id) REFERENCES pedidos(id),
    FOREIGN KEY (produto_id) REFERENCES produtos(id)
);
